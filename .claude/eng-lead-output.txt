```python
@dataclass
class FileEvidence:
    """Per-file evidence tracking."""
    path: str
    first_read: float = 0.0       # timestamp of first Read
    last_modified: float = 0.0    # timestamp of last Write/Edit
    last_verified: float = 0.0    # timestamp of last verification
    read_count: int = 0
    modify_count: int = 0
    verify_count: int = 0

    @property
    def state(self) -> FileState:
        if self.last_verified > self.last_modified > 0:
            return FileState.VERIFIED
        if self.last_modified > 0:
            return FileState.MODIFIED
        return FileState.READ

    @property
    def was_read_before_write(self) -> bool:
        """True if file was read before first modification."""
        if self.last_modified == 0:
            return True  # never modified, no violation
        return self.first_read > 0 and self.first_read < self.last_modified

def detect_phase_signal(text: str) -> tuple[Phase | None, float]:
    """Detect strongest phase signal from text. Returns (phase, confidence)."""
    text_lower = text.lower()
    scores: dict[Phase, float] = {}
    for phase, config in PHASE_SIGNALS.items():
        keywords = config["keywords"]
        weight = float(config.get("weight", 1.0))
        hits = sum(1 for kw in keywords if kw in text_lower)
        if hits > 0:
            scores[phase] = hits * weight
    if not scores:
        return None, 0.0
    best = max(scores, key=scores.get)
    total_hits = sum(scores.values())
    confidence = scores[best] / total_hits if total_hits > 0 else 0.0
    return best, confidence


def attempt_transition(state: RuntimeState, target: Phase, trigger: str) -> tuple[bool, str]:
    """Attempt a GOLDE phase transition. Returns (success, message)."""
    current = state.current_phase
    legal = LEGAL_TRANSITIONS.get(current, set())

    transition = PhaseTransition(
        from_phase=current,
        to_phase=target,
        timestamp=time.time(),
        trigger=trigger,
        legal=target in legal,
    )

    # Record phase duration
    if state.phase_entry_time > 0:
        duration = time.time() - state.phase_entry_time
        phase_key = current.value
        state.phase_durations[phase_key] = state.phase_durations.get(phase_key, 0.0) + duration

    if target in legal:
        state.current_phase = target
        state.phase_entry_time = time.time()
        state.phase_history.append(asdict(transition))
        guidance = PHASE_SIGNALS.get(target, {}).get("guidance", "")
        return True, str(guidance)
    else:
        # Illegal transition — inject warning but allow with correction
        state.phase_history.append(asdict(transition))
        warning = (
            f"[GOLDE:TRANSITION WARNING] {current.value} -> {target.value} is unusual.\n"
            f"Expected transitions from {current.value}: {', '.join(p.value for p in legal)}.\n"
        )
        if current == Phase.IDLE and target == Phase.EXECUTE:
            # Common shortcut — allow but suggest orient first
            state.current_phase = target
            state.phase_entry_time = time.time()
            warning += "Hint: Consider a quick ORIENT phase first (Read Before Write)."
        elif current == Phase.EXECUTE and target == Phase.ORIENT:
            # Going back — this is fine, it means they realized they need more context
            state.current_phase = target
            state.phase_entry_time = time.time()
            warning = str(PHASE_SIGNALS[Phase.ORIENT]["guidance"])
        else:
            state.current_phase = target
            state.phase_entry_time = time.time()
        return False, warning


# ─── Evidence Tracker ───────────────────────────────────────────────

def track_file_read(state: RuntimeState, file_path: str) -> None:
    """Record a file read event."""
    normalized = os.path.normpath(file_path)
    if normalized not in state.files:
        state.files[normalized] = asdict(FileEvidence(path=normalized))
    entry = state.files[normalized]
    if entry["first_read"] == 0.0:
        entry["first_read"] = time.time()
    entry["read_count"] = entry.get("read_count", 0) + 1


def track_file_modify(state: RuntimeState, file_path: str) -> str:
    """Record a file modification. Returns warning if Read-Before-Write violated."""
    normalized = os.path.normpath(file_path)
    now = time.time()
    warning = ""

    if normalized not in state.files:
        state.files[normalized] = asdict(FileEvidence(path=normalized))

    entry = state.files[normalized]
    entry["last_modified"] = now
    entry["modify_count"] = entry.get("modify_count", 0) + 1
    state.total_modifications += 1

    # Read-Before-Write check
    if entry.get("first_read", 0.0) == 0.0:
        state.read_before_write_violations += 1
        warning = (
            f"[READ-BEFORE-WRITE VIOLATION] Modifying '{os.path.basename(normalized)}' without reading it first.\n"
            "Principle: Never modify code you haven't read."
        )

    # Track unverified files
    if normalized not in state.unverified_files:
        state.unverified_files.append(normalized)

    return warning


def track_file_verify(state: RuntimeState, file_paths: list[str]) -> None:
    """Record verification for file(s)."""
    now = time.time()
    for fp in file_paths:
        normalized = os.path.normpath(fp)
        if normalized in state.files:
            state.files[normalized]["last_verified"] = now
            state.files[normalized]["verify_count"] = state.files[normalized].get("verify_count", 0) + 1
        if normalized in state.unverified_files:
            state.unverified_files.remove(normalized)
    state.verification_runs += 1


# ─── Cost Tracker ───────────────────────────────────────────────────

def classify_bash_tier(command: str) -> ToolTier:
    """Classify a Bash command into a cost tier."""
    cmd_lower = command.lower().strip()
    for pattern in BASH_READ_PATTERNS:
        if cmd_lower.startswith(pattern):
            return ToolTier.FREE
    for pattern in BASH_VERIFY_PATTERNS:
        if any(p in cmd_lower for p in [pattern]):
            return ToolTier.LOW  # verification is LOW cost, not MEDIUM
    return ToolTier.MEDIUM


def is_bash_verification(command: str) -> bool:
    """Check if a Bash command is a verification action."""
    cmd_lower = command.lower()
    return any(p in cmd_lower for p in BASH_VERIFY_PATTERNS)


def track_tool_cost(state: RuntimeState, tool_name: str, tool_input: dict) -> str:
    """Track tool cost and return warning if cheapest-first violated."""
    state.total_tool_calls += 1
    state.tool_counts[tool_name] = state.tool_counts.get(tool_name, 0) + 1

    # Determine tier
    tier = TOOL_TIERS.get(tool_name, ToolTier.MEDIUM)
    if tool_name == "Bash":
        command = tool_input.get("command", "")
        tier = classify_bash_tier(command)

    tier_key = tier.value
    state.tier_counts[tier_key] = state.tier_counts.get(tier_key, 0) + 1

    # Cheapest-first check: warn if using HIGH/EXPENSIVE tools
    # in ORIENT or ANALYZE phases (should be reading, not writing)
    warning = ""
    if tier in (ToolTier.HIGH, ToolTier.EXPENSIVE):
        if state.current_phase in (Phase.ORIENT, Phase.ANALYZE):
            state.cheapest_first_violations += 1
            state.cost_warnings += 1
            warning = (
                f"[CHEAPEST-FIRST] Using {tier.value}-cost tool '{tool_name}' during {state.current_phase.value} phase.\n"
                "During Orient/Analyze, prefer: Glob/Grep(free) -> Read(low) -> Bash-readonly(free).\n"
                "Save writes and agents for Execute phase."
            )

    return warning

def score_agent_prompt(prompt: str) -> tuple[float, list[str], list[str]]:
    """Score agent prompt quality 0.0-1.0.
    Returns (score, strengths, weaknesses).
    """
    prompt_lower = prompt.lower()
    score = 0.0
    strengths: list[str] = []
    weaknesses: list[str] = []

    # Check anti-patterns (each one subtracts 0.3)
    for pattern in DELEGATION_ANTI_PATTERNS:
        if pattern in prompt_lower:
            score -= 0.3
            weaknesses.append(f"Delegation anti-pattern: '{pattern}'")

    # Check positive indicators
    for indicator_key, config in QUALITY_INDICATORS.items():
        found = any(p in prompt_lower for p in config["patterns"])
        if found:
            score += config["weight"]
            strengths.append(config["label"])
        else:
            if config["weight"] >= 0.2:
                weaknesses.append(f"Missing: {config['label']}")

    # Length penalty: very short prompts (<50 chars) are likely vague
    if len(prompt) < 50:
        score -= 0.1
        weaknesses.append("Prompt too short for meaningful delegation")

    # Clamp
    score = max(0.0, min(1.0, score))
    return score, strengths, weaknesses


def check_agent_quality(state: RuntimeState, prompt: str) -> tuple[bool, str]:
    """Full agent quality check. Returns (passed, message)."""
    score, strengths, weaknesses = score_agent_prompt(prompt)

    has_anti_pattern = any(p in prompt.lower() for p in DELEGATION_ANTI_PATTERNS)
    has_file_refs = any(
        p in prompt.lower() for p in QUALITY_INDICATORS["file_ref"]["patterns"]
    )
    has_specific = any(
        p in prompt.lower() for p in QUALITY_INDICATORS["specific_action"]["patterns"]
    )

    call = AgentCall(
        timestamp=time.time(),
        prompt_length=len(prompt),
        has_file_refs=has_file_refs,
        has_specific_action=has_specific,
        has_anti_pattern=has_anti_pattern,
        quality_score=score,
    )
    state.agent_calls.append(asdict(call))
    state.total_agent_calls += 1

    # Running average
    scores = [c["quality_score"] for c in state.agent_calls]
    state.avg_agent_quality = sum(scores) / len(scores) if scores else 0.0

    # Build response
    if score < 0.3:
        state.quality_gate_blocks += 1
        parts = [
            f"[QUALITY GATE] Score: {score:.1f}/1.0 — Never Delegate Understanding.",
        ]
        if weaknesses:
            parts.append("Issues: " + "; ".join(weaknesses[:3]))
        parts.append(
            "Fix: Synthesize understanding yourself, then delegate with file:line specs."
        )
        return False, "\n".join(parts)
    elif score < 0.6:
        parts = [f"[QUALITY HINT] Score: {score:.1f}/1.0."]
        if weaknesses:
            parts.append("Could improve: " + "; ".join(weaknesses[:2]))
        return True, "\n".join(parts)
    else:
        return True, ""  # Good quality, no injection needed


# ─── Verification Gate ──────────────────────────────────────────────

def build_verification_report(state: RuntimeState) -> str:
    """Build a gap report showing unverified modifications."""
    unverified = state.unverified_files
    total_mods = state.total_modifications
    verified_count = state.verification_runs

    if not unverified:
        return ""

    count = len(unverified)

    if count >= 5:
        # Full gap report
        file_list = "\n".join(f"  - {os.path.basename(f)}" for f in unverified[:8])
        if count > 8:
            file_list += f"\n  ... and {count - 8} more"
        return (
            f"[VERIFICATION GAP] {count} files modified without verification.\n"
            f"Modified: {total_mods} | Verified: {verified_count} | Coverage: {max(0, total_mods - count)}/{total_mods}\n"
            f"Unverified files:\n{file_list}\n"
            "Action: Run tests, type check, or lint to verify these changes."
        )
    elif count >= 3:
        return (
            f"[VERIFY HINT] {count} files modified since last verification.\n"
            "Consider running a quick check before continuing."
        )
    return ""


# ─── Metrics Computation ───────────────────────────────────────────

def compute_metrics(state: RuntimeState) -> dict:
    """Compute session metrics from current state."""
    # Evidence coverage
    total_files = len(state.files)
    modified_files = sum(1 for f in state.files.values() if f.get("modify_count", 0) > 0)
    verified_files = sum(
        1 for f in state.files.values()
        if f.get("last_verified", 0) > f.get("last_modified", 0) > 0
    )
    evidence_coverage = verified_files / modified_files if modified_files > 0 else 1.0

    # Read-before-write compliance
    rbw_eligible = sum(1 for f in state.files.values() if f.get("modify_count", 0) > 0)
    rbw_compliant = sum(
        1 for f in state.files.values()
        if f.get("modify_count", 0) > 0
        and f.get("first_read", 0) > 0
        and f.get("first_read", 0) < f.get("last_modified", 0)
    )
    rbw_rate = rbw_compliant / rbw_eligible if rbw_eligible > 0 else 1.0

    # Cost efficiency
    total_tools = state.total_tool_calls
    expensive_tools = state.tier_counts.get("high", 0) + state.tier_counts.get("expensive", 0)
    cost_efficiency = 1.0 - (expensive_tools / total_tools) if total_tools > 0 else 1.0

    # GOLDE compliance (did we visit orient before execute?)
    phase_order = [p.get("to_phase", "") for p in state.phase_history]
    golde_compliant = True
    for i, phase in enumerate(phase_order):
        if phase == "execute" and "orient" not in phase_order[:i]:
            golde_compliant = False
            break

    # Verification completeness
    verify_completeness = evidence_coverage  # same metric, different name for clarity

    return {
        "session_id": state.session_id,
        "timestamp": time.time(),
        "duration_seconds": time.time() - state.session_start if state.session_start > 0 else 0,

        # Principle compliance
        "golde_compliant": golde_compliant,
        "read_before_write_rate": round(rbw_rate, 3),
        "cheapest_first_violations": state.cheapest_first_violations,
        "phase_coverage": list(set(p.get("to_phase", "") for p in state.phase_history)),

        # Evidence coverage
        "files_tracked": total_files,
        "files_modified": modified_files,
        "files_verified": verified_files,
        "evidence_coverage": round(evidence_coverage, 3),

        # Cost efficiency
        "total_tool_calls": total_tools,
        "tier_distribution": dict(state.tier_counts),
        "cost_efficiency": round(cost_efficiency, 3),
        "cost_warnings": state.cost_warnings,

        # Agent quality
        "total_agent_calls": state.total_agent_calls,
        "avg_agent_quality": round(state.avg_agent_quality, 3),
        "quality_gate_blocks": state.quality_gate_blocks,

        # Verification
        "verification_runs": state.verification_runs,
        "unverified_files": len(state.unverified_files),
        "verify_completeness": round(verify_completeness, 3),

        # Composite score: weighted average of key metrics
        "composite_score": round(
            rbw_rate * 0.2
            + evidence_coverage * 0.25
            + cost_efficiency * 0.15
            + state.avg_agent_quality * 0.15
            + verify_completeness * 0.25,
            3
        ),
    }


# ─── State Persistence ──────────────────────────────────────────────

def load_state() -> RuntimeState:
    try:
        if RUNTIME_STATE.exists():
            data = json.loads(RUNTIME_STATE.read_text())
            return RuntimeState.from_json(data)
    except (json.JSONDecodeError, OSError, TypeError):
        pass

    session_id = hashlib.md5(str(time.time()).encode()).hexdigest()[:8]
    return RuntimeState(session_id=session_id, session_start=time.time())


def save_state(state: RuntimeState) -> None:
    try:
        STATE_DIR.mkdir(parents=True, exist_ok=True)
        RUNTIME_STATE.write_text(state.to_json())
    except OSError:
        pass


def save_metrics(state: RuntimeState) -> None:
    """Persist session metrics to metrics.json and append to sessions.jsonl."""
    metrics = compute_metrics(state)
    try:
        STATE_DIR.mkdir(parents=True, exist_ok=True)
        METRICS_FILE.write_text(json.dumps(metrics, indent=2))
        with open(SESSION_LOG, "a") as f:
            f.write(json.dumps(metrics) + "\n")
    except OSError:
        pass

def select_skills_for_context(phase: Phase, turn_count: int) -> str:
    """Select and format compressed skills relevant to current context."""
    selected = []
    phase_val = phase.value

    for name, skill in SKILL_COMPACT.items():
        phases = str(skill["phases"])
        if phases == "all" or phase_val in phases.split(","):
            selected.append((int(skill["priority"]), name, str(skill["compact"])))

    # Sort by priority, take top 4 to stay within budget
    selected.sort(key=lambda x: x[0])
    max_skills = 3 if turn_count > 10 else 4  # fewer skills as context grows

    parts = []
    for _, name, compact in selected[:max_skills]:
        parts.append(compact)

    if not parts:
        return ""

    return "[CC-PATH SKILLS]\n" + "\n---\n".join(parts)


# ─── Main Hook Handler ──────────────────────────────────────────────

def handle_user_prompt_submit(hook_input: dict, state: RuntimeState) -> dict:
    """Process UserPromptSubmit: GOLDE phase detection + skill injection."""
    content = hook_input.get("content", "")
    if isinstance(content, list):
        content = " ".join(b.get("text", "") for b in content if isinstance(b, dict))
    if not content:
        return {}

    state.turn_count += 1

    # Detect phase signal
    target_phase, confidence = detect_phase_signal(str(content))
    if target_phase is None:
        return {}

    # Only transition if confidence is reasonable
    if confidence < 0.3 and state.current_phase != Phase.IDLE:
        return {}  # weak signal, stay in current phase

    # Attempt transition
    _, guidance = attempt_transition(state, target_phase, f"user_prompt:{content[:50]}")

    # Select relevant skills
    skill_context = select_skills_for_context(state.current_phase, state.turn_count)

    # Build injection
    parts = [p for p in [guidance, skill_context] if p]
    if parts:
        # Add verification gap report if in EXECUTE phase with pending verifications
        if state.current_phase == Phase.EXECUTE:
            gap = build_verification_report(state)
            if gap:
                parts.append(gap)

        save_state(state)
        return {"additionalContext": "\n\n".join(parts)}

    save_state(state)
    return {}


def handle_pre_tool_use(hook_input: dict, state: RuntimeState) -> dict:
    """Process PreToolUse: cost tracking + agent quality gate."""
    tool_name = hook_input.get("tool_name", "")
    tool_input = hook_input.get("tool_input", {})

    # Cost tracking
    cost_warning = track_tool_cost(state, tool_name, tool_input)

    # Agent quality gate
    agent_message = ""
    if tool_name == "Agent":
        prompt = tool_input.get("prompt", "")
        _, agent_message = check_agent_quality(state, prompt)

    parts = [p for p in [cost_warning, agent_message] if p]
    save_state(state)

    if parts:
        return {"additionalContext": "\n".join(parts)}
    return {}


def handle_post_tool_use(hook_input: dict, state: RuntimeState) -> dict:
    """Process PostToolUse: evidence tracking + verification gate."""
    tool_name = hook_input.get("tool_name", "")
    tool_input = hook_input.get("tool_input", {})
    messages: list[str] = []

    # File read tracking
    if tool_name == "Read":
        file_path = tool_input.get("file_path", "")
        if file_path:
            track_file_read(state, file_path)

    # File modification tracking
    if tool_name in ("Write", "Edit", "MultiEdit", "NotebookEdit"):
        file_path = tool_input.get("file_path", "")
        if file_path:
            warning = track_file_modify(state, file_path)
            if warning:
                messages.append(warning)

    # Bash verification detection
    if tool_name == "Bash":
        command = tool_input.get("command", "")
        if is_bash_verification(command):
            # Mark all currently unverified files as verified
            track_file_verify(state, list(state.unverified_files))
            # Auto-transition to VERIFY phase if in EXECUTE
            if state.current_phase == Phase.EXECUTE:
                attempt_transition(state, Phase.VERIFY, f"bash_verify:{command[:30]}")

    # Verification gap report (on every write/edit)
    if tool_name in ("Write", "Edit", "MultiEdit", "NotebookEdit"):
        gap = build_verification_report(state)
        if gap:
            messages.append(gap)

    # Periodic metrics save (every 20 tool calls)
    if state.total_tool_calls % 20 == 0:
        save_metrics(state)

    save_state(state)
    if messages:
        return {"additionalContext": "\n\n".join(messages)}
    return {}


def handle_notification(hook_input: dict, state: RuntimeState) -> dict:
    """Process Notification: session end metrics."""
    # Save final metrics on stop
    save_metrics(state)
    save_state(state)
    return {}


def main() -> None:
    try:
        hook_input = json.loads(sys.stdin.read())
    except (json.JSONDecodeError, EOFError):
        print(json.dumps({}))
        return

    hook_event = os.environ.get("CLAUDE_HOOK_EVENT", "")
    state = load_state()

    handlers = {
        "UserPromptSubmit": handle_user_prompt_submit,
        "PreToolUse": handle_pre_tool_use,
        "PostToolUse": handle_post_tool_use,
        "Notification": handle_notification,
    }

    handler = handlers.get(hook_event)
    if handler:
        result = handler(hook_input, state)
        print(json.dumps(result))
    else:
        print(json.dumps({}))


if __name__ == "__main__":
    main()
```

### 1.3 Hook Event Flow

```
UserPromptSubmit
  1. Parse user text from content (string or block array)
  2. Increment turn_count
  3. detect_phase_signal() -> (target_phase, confidence)
  4. If confidence >= 0.3: attempt_transition() enforces LEGAL_TRANSITIONS
  5. select_skills_for_context() picks top 3-4 compressed skills by priority+phase
  6. If in EXECUTE: append build_verification_report()
  7. Inject combined guidance + skills + gap report

PreToolUse
  1. track_tool_cost() -> classify tool tier, check cheapest-first violation
  2. If tool is Agent: score_agent_prompt() -> (score, strengths, weaknesses)
     - score < 0.3 -> inject quality gate block message
     - score 0.3-0.6 -> inject quality hint
     - score >= 0.6 -> no injection
  3. Inject combined cost warning + agent message

PostToolUse
  1. If Read: track_file_read() (record timestamp, increment count)
  2. If Write/Edit/MultiEdit: track_file_modify()
     - Check read-before-write: was this file read first?
     - Add to unverified_files list
  3. If Bash + is_bash_verification(): 
     - track_file_verify() for all unverified files
     - Auto-transition EXECUTE -> VERIFY
  4. Every 20 tool calls: save_metrics() to disk
  5. Inject read-before-write warning + verification gap report
```

## Deliverable 2: Compact Skill Injections

### 2.1 Design Rationale

The 18 SKILL.md files total ~109KB (~27K tokens). Even a single skill like `anthropic-prompt-craft` at 7.7KB (~1,900 tokens) exceeds what should be injected per turn. The solution is embedded in the Python code above as `SKILL_COMPACT` dict.

**Compression strategy:**
- Each skill reduced to **a single string under 150 tokens** (~500-600 chars)
- Format: **telegram-style declarative commands** -- no markdown, no headers, no bullets, no prose. Just patterns and rules
- Structure that survives compaction: `LABEL: numbered or arrow-linked rules. Key terms in caps.`
- This format works because the model interprets capitalized terms as strong signals and numbered lists as ordered procedures

### 2.2 Priority Ranking

| Priority | Skills | Inject when | Rationale |
|----------|--------|-------------|-----------|
| 1 (always) | `principles`, `verify` | Every phase / verify phase | Core behavioral contract |
| 2 (often) | `problem-solve`, `architecture`, `strategic-plan`, `agent-mastery` | Phase-matched | High-impact methodology |
| 3 (sometimes) | `token-zero`, `context-engine`, `research`, `multi-agent` | Phase-matched | Efficiency + orchestration |
| 4 (rarely) | `prompt-craft`, `document-craft`, `agent-interconnect`, `harness-craft` | Phase-matched | Specialized craft |
| 5 (on-demand) | `agent-automation`, `skill-forge`, `folder-mastery` | Phase-matched | Rare use cases |

### 2.3 Situational Selection

The `select_skills_for_context()` function in the code above implements this:
- Filter skills whose `phases` field includes current GOLDE phase (or "all")
- Sort by priority (1 first)
- Take top **4 skills** in early session (turn_count <= 10), **3 skills** later (to save tokens as context grows)
- Budget: ~600 tokens total injection per turn (4 skills x 150 tokens)

### 2.4 Compressed vs Full Examples

**`anthropic-verify` -- Full: 5,160 bytes (~1,290 tokens)**

Contains 6 security principles, 7 permission modes table, circuit breaker code snippets, Git safety detection lists, file history/undo mechanics, full checklist template.

**`anthropic-verify` -- Compressed: ~130 tokens**
```
VERIFY: Independent adversarial verification required. Author!=Verifier. Fail-Closed:unknown=dangerous. Checklist: original problem solved? existing tests pass? new tests cover change? no unintended side-effects? FACT-only conclusions? Circuit breaker: 3 consecutive fails->stop strategy.
```

**`anthropic-architecture` -- Full: 6,129 bytes (~1,532 tokens)**

7-layer pipeline, 6 design patterns table, Fail-Closed code example, 35-line store code, 3-vector extension table, import cycle prevention code, feature flags, directory structure rules, AsyncGenerator pipeline.

**`anthropic-architecture` -- Compressed: ~115 tokens**
```
ARCH: 7-layer pipeline(deps flow down only). Fail-Closed defaults: isConcurrencySafe:false, isReadOnly:false until proven. 35-line pub/sub store>Redux. Builder/Factory for safe defaults. Import cycles: types/ extraction, lazy require, registry pattern. Directory promotion: 3+ files or UI/prompt separation.
```

**Compression ratio: ~10-12x with ~80% behavioral signal retention.** The full versions contain examples and code snippets that help understanding but don't change model behavior. The compressed versions retain the decision rules and action triggers.

### Design Decisions for Settings

**Why separate matchers instead of one big regex:**
1. `Agent|Task` needs 5s timeout (quality scoring is heavier). File writes need only 3s. Reads need only 2s. Separate matchers = separate timeouts.
2. The `Read` PostToolUse hook fires on every file read -- must be ultra-fast (2s timeout). If bundled with Write hooks, a slow verification gap report could delay Read tracking.
3. Debugging: separate matchers produce separate `statusMessage` strings, making it clear which hook is running.

**Why NOT hook Glob/Grep:**
These are FREE-tier tools that fire extremely frequently. Hooking them would add 2-3s latency to every search operation for zero behavioral value. The cost tracker counts them via the PreToolUse on Agent/Write/etc. -- we know they exist from the state's tool_counts, but we don't need to intercept them.

**Why add a Stop hook:**
The v1.0 had no session-end metrics capture. The Stop hook writes final metrics to `metrics.json` and appends to `sessions.jsonl` for cross-session trend analysis.

## Deliverable 4: Updated CLAUDE.md

### 4.1 What Actually Changes Behavior

Based on analysis of the current files, three things matter for behavior change:

1. **Imperative commands with consequences** -- "NEVER modify code without reading it first" changes behavior. "Read Before Write is a principle" does not.
2. **Concrete decision rules** -- "If 3+ files modified without tests, STOP and verify" is actionable. "Verification is important" is not.
3. **Scarcity** -- A 50-line CLAUDE.md outperforms a 200-line one because the model treats shorter instructions as higher priority and doesn't compaction-strip them.

The current CLAUDE.md (53 lines) and .claude/CLAUDE.md (39 lines) both list the 10 principles as passive knowledge. The v2.0 versions should be **commands, not descriptions**.

### 4.2 Minimum Effective CLAUDE.md (root)

```markdown
# cc-path — Active Enforcement Runtime

Runtime hooks ENFORCE these rules automatically. Violations are detected and flagged.

## MANDATORY BEHAVIOR (hooks will warn on violation)

1. NEVER modify a file you haven't Read first. The hook tracks every Read and Write.
2. NEVER use expensive tools (Write/Edit/Agent/WebSearch) during Orient or Analyze phases. Use Glob/Grep/Read first.
3. NEVER delegate to Agent without file:line references in the prompt. The quality gate scores your prompt 0-1.0 and warns below 0.6.
4. AFTER 3+ file modifications, run tests or verification before continuing. The hook counts and will escalate warnings.
5. WHEN a Bash verification command runs, the hook resets the verification counter. Use: test, pytest, tsc, lint, etc.

## GOLDE CYCLE (hook auto-detects and transitions)

Orient(read) -> Analyze(hypothesize) -> Plan(decompose) -> Execute(minimum change) -> Verify(evidence) -> Learn(preserve non-obvious)

Illegal transitions are warned. The state machine tracks where you are.

## COST TIERS (hook tracks)

FREE: Glob, Grep | LOW: Read, Bash(readonly) | MEDIUM: Bash(mutate) | HIGH: Write, Edit | EXPENSIVE: Agent, WebSearch

Cheapest-First: exhaust lower tiers before using higher ones.

## SKILL INJECTION

Compressed skills auto-inject based on GOLDE phase. No manual invocation needed.
Full skills available as `/anthropic-*` when deeper reference is needed.
```

### 4.3 Minimum Effective `.claude/CLAUDE.md` (project-level)

```markdown

Hooks: GOLDE phase tracking, agent quality gate, evidence tracking, cost tracking, verification enforcement.
State: `.cc-path/runtime.json` | Metrics: `.cc-path/metrics.json` | History: `.cc-path/sessions.jsonl`

## Hook Behavior Summary

| Event | What Happens | Timeout |
|-------|-------------|---------|

| PreToolUse:Write | Cost tier classification + phase violation check | 3s |

| PostToolUse:Write | Read-before-write check + unverified file tracking | 3s |
```

### 4.4 Making Principles Sticky Across Compaction

The key insight: during context compaction, the model summarizes conversation history but preserves system prompt and CLAUDE.md verbatim (they're re-injected). Therefore:

1. **Put behavioral rules in CLAUDE.md, not in conversation** -- conversation gets compacted, CLAUDE.md doesn't
2. **Use numbered rules with CAPS verbs** -- "NEVER", "ALWAYS", "AFTER N" survive compaction summaries better than prose
3. **Reference the hook system** -- "the hook tracks" and "the hook will warn" creates accountability because the model knows it will be caught
4. **Keep it under 60 lines** -- the entire CLAUDE.md should fit in ~800 tokens so it never gets trimmed

### 5.2 Metric Computation Details

| Metric | Formula | What it Measures |
|--------|---------|-----------------|
| `read_before_write_rate` | `count(files where first_read < last_modified) / count(files where modify_count > 0)` | Principle #1 compliance |
| `evidence_coverage` | `count(files where last_verified > last_modified) / count(files where modify_count > 0)` | Principle #5 compliance |
| `cost_efficiency` | `1.0 - (tier_counts.high + tier_counts.expensive) / total_tool_calls` | Principle #6 compliance |
| `avg_agent_quality` | `mean(agent_calls[*].quality_score)` | Principle #8 compliance |
| `verify_completeness` | Same as evidence_coverage | Principle #5 (verification angle) |
| `composite_score` | `rbw*0.2 + evidence*0.25 + cost*0.15 + agent_quality*0.15 + verify*0.25` | Overall session quality |

**Weight rationale**: Verification (evidence + verify = 0.50 total) is weighted heaviest because "Verify Before Claiming" is the most commonly violated principle and has the highest impact on output quality. Read-before-write (0.20) catches planning-phase discipline. Cost efficiency (0.15) and agent quality (0.15) are enablers, not outcomes.

## Trade-offs

| Decision | Pros | Cons |
|----------|------|------|
| Embedded SKILL_COMPACT in Python instead of separate files | Single deployment artifact, no file I/O per turn, instant access | Skills can't be updated without redeploying the hook, harder for non-developers to edit |
| Per-file evidence tracking | Precise read-before-write detection, per-file verification gaps | State file grows linearly with files touched (~200 bytes per file entry), may slow I/O in very large sessions |
| GOLDE state machine with legal transitions | Prevents incoherent phase jumps, enforces discipline | May occasionally over-constrain the model (e.g., quick fix without orient). Mitigated by IDLE->EXECUTE shortcut |
| Keyword-based phase detection (improved but still keyword) | Zero additional cost, works in 2s timeout, bilingual | Can still misclassify -- "check" in "check out this pattern" would match VERIFY. Mitigated by confidence threshold (0.3) and legal transition enforcement |
| Composite score as weighted average | Single number for quick assessment, comparable across sessions | Weights are opinionated -- different teams might value cost_efficiency vs verify_completeness differently. Consider making weights configurable. |
| 3-4 skills injected per turn (~600 tokens) | Manageable token budget, phase-relevant | Drops 14 of 18 skills per turn. Mitigated by priority ranking and full skills still available via `/anthropic-*` |
| Separate matcher entries in settings.json | Precise timeouts, clear status messages, debuggable | More JSON to maintain, 6 hook registrations instead of 3. But hook overhead is only paid when the matcher matches. |
